mindctl a un agent "depguard" qui lit docker-compose.yml
et extrait automatiquement avec ce code bash exact :

DB_TYPE=$(grep -i "image:" docker-compose.yml \
          | grep -iE "postgres|mysql|mongo" \
          | grep -oiE "postgres|mysql|mongo" | head -1)

DB_PORT=$(grep -A10 "$DB_TYPE" docker-compose.yml \
          | grep -oP '\d+:\d+' | head -1 | cut -d: -f1)

DB_USER=$(grep -iE "POSTGRES_USER|MYSQL_USER" \
          docker-compose.yml \
          | cut -d= -f2 | tr -d '"' | head -1)

DB_PASS=$(grep -iE "POSTGRES_PASSWORD|MYSQL_PASSWORD" \
          docker-compose.yml \
          | cut -d= -f2 | tr -d '"' | head -1)

DB_NAME=$(grep -iE "POSTGRES_DB|MYSQL_DATABASE" \
          docker-compose.yml \
          | cut -d= -f2 | tr -d '"' | head -1)
