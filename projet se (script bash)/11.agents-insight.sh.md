```bash
#!/bin/bash

source ~/mindctl/mindctl.conf
source ~/mindctl/core/logger.sh
source ~/mindctl/core/errors.sh
source ~/mindctl/core/llm.sh

RAPPORT_DEPGUARD="/var/log/mindctl/reports/last_depguard.txt"
RAPPORT_DATA="/var/log/mindctl/reports/last_data.txt"

log_message "INFOS" "insight démarré"
echo ""
echo "[insight] ── Corrélation croisée système + données ──"

# ── Vérifier que les rapports existent ───────────────────
if [ ! -f "$RAPPORT_DEPGUARD" ]; then
    log_message "ERROR" "Rapport depguard manquant — lancez depguard d'abord"
    erreur 102
fi

if [ ! -f "$RAPPORT_DATA" ]; then
    log_message "ERROR" "Rapport données manquant — lancez cleaner et analyzer d'abord"
    erreur 102
fi

# ── Lire les deux rapports ────────────────────────────────
RESULTATS_SYSTEME=$(cat "$RAPPORT_DEPGUARD")
RESULTATS_DONNEES=$(cat "$RAPPORT_DATA")

echo "[insight] Rapport depguard chargé ✅"
echo "[insight] Rapport données chargé ✅"
echo ""

# ── Vérifier que les rapports ne sont pas vides ───────────
if [ -z "$RESULTATS_SYSTEME" ]; then
    log_message "ERROR" "Rapport depguard vide"
    erreur 107
fi

if [ -z "$RESULTATS_DONNEES" ]; then
    log_message "ERROR" "Rapport données vide"
    erreur 107
fi

# ── Construire le prompt de corrélation ──────────────────
echo "[insight] Analyse de corrélation en cours..."
echo ""

PROMPT="Tu es un expert en systèmes Linux et en analyse de données.

Voici les anomalies détectées dans la CONFIGURATION SYSTÈME par depguard :
$RESULTATS_SYSTEME

Voici les anomalies détectées dans les DONNÉES DE PRODUCTION :
$RESULTATS_DONNEES

Analyse ces deux sources et réponds en français :

1. Y a-t-il une corrélation entre les anomalies système et les anomalies données ?
2. Si oui, explique le lien de cause à effet en une phrase claire.
3. Quelle anomalie est la plus critique à résoudre en priorité ?
4. Donne 3 recommandations concrètes et actionnables.

Sois précis et concis — maximum 10 lignes au total."

# ── Envoyer au LLM ────────────────────────────────────────
CORRELATION=$(appeler_llm "$PROMPT")

echo "[insight] Résultat de la corrélation :"
echo "══════════════════════════════════════"
echo "$CORRELATION"
echo "══════════════════════════════════════"
echo ""

# ── Sauvegarder le rapport final ─────────────────────────
RAPPORT_FINAL="/var/log/mindctl/reports/insight_$(date +%Y-%m-%d-%H-%M-%S).txt"

cat > "$RAPPORT_FINAL" << EOF
[insight] Rapport de corrélation — $(date +"%Y-%m-%d %H:%M:%S")

── Anomalies système (depguard) ──
$RESULTATS_SYSTEME

── Anomalies données ──
$RESULTATS_DONNEES

── Corrélation LLM ──
$CORRELATION
EOF

echo "[insight] Rapport sauvegardé → $RAPPORT_FINAL ✅"
log_message "INFOS" "Rapport insight sauvegardé → $RAPPORT_FINAL"
log_message "INFOS" "insight terminé"
```

Sauvegarder : `Ctrl+O` puis `Entrée` puis `Ctrl+X`
