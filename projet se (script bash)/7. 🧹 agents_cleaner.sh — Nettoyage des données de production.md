## 🎯 Ce que fait ce fichier

Se connecte directement à la base de données détectée par depguard, extrait les données, et les nettoie intelligemment :

```
Ligne vide              →  supprimer
Doublon                 →  supprimer
Valeur numérique nulle  →  remplacer par la MOYENNE de la colonne
Valeur string nulle     →  remplacer par la valeur la PLUS FRÉQUENTE
```

> [!info] Lit le contexte `/tmp/mindctl_context.env` écrit par depguard pour se connecter à la base sans configuration manuelle

## 💻 Le code complet

```bash
#!/bin/bash

source ~/mindctl/mindctl.conf
source ~/mindctl/core/logger.sh
source ~/mindctl/core/errors.sh

CONTEXT_FILE="/tmp/mindctl_context.env"

# ── Vérifier le contexte depguard ────────────────────────
if [ ! -f "$CONTEXT_FILE" ]; then
    log_message "ERROR" "Contexte depguard manquant"
    erreur 102
fi

source "$CONTEXT_FILE"

# ── Vérifier que la base est disponible ──────────────────
if [ -z "$DB_TYPE" ]; then
    log_message "ERROR" "Aucune base de données détectée"
    erreur 107
fi

log_message "INFOS" "cleaner démarré — base : $DB_TYPE sur port $DB_PORT"
echo ""
echo "[cleaner] ── Nettoyage des données de production ──"
echo "[cleaner] Base : $DB_TYPE | Port : $DB_PORT | Base : $DB_NAME"

# ── Récupérer la liste des tables ────────────────────────
function get_tables() {
    if [ "$DB_TYPE" = "postgres" ]; then
        PGPASSWORD=$DB_PASS psql -h localhost -p $DB_PORT \
            -U $DB_USER -d $DB_NAME -t -c \
            "SELECT tablename FROM pg_tables WHERE schemaname='public';" \
            2>/dev/null | grep -v '^$' | xargs
    elif [ "$DB_TYPE" = "mysql" ]; then
        mysql -h localhost -P $DB_PORT \
            -u $DB_USER -p$DB_PASS $DB_NAME \
            -e "SHOW TABLES;" 2>/dev/null | tail -n +2 | xargs
    fi
}

# ── Extraire une table en CSV ─────────────────────────────
function extraire_table() {
    local TABLE="$1"
    local FICHIER_CSV="/tmp/mindctl_${TABLE}.csv"

    log_message "INFOS" "Extraction de la table $TABLE"

    if [ "$DB_TYPE" = "postgres" ]; then
        PGPASSWORD=$DB_PASS psql -h localhost -p $DB_PORT \
            -U $DB_USER -d $DB_NAME -c \
            "\COPY $TABLE TO '$FICHIER_CSV' CSV HEADER" \
            2>/dev/null
    elif [ "$DB_TYPE" = "mysql" ]; then
        mysql -h localhost -P $DB_PORT \
            -u $DB_USER -p$DB_PASS $DB_NAME \
            -e "SELECT * FROM $TABLE INTO OUTFILE '$FICHIER_CSV'
                FIELDS TERMINATED BY ','
                ENCLOSED BY '\"'
                LINES TERMINATED BY '\n';" \
            2>/dev/null
    fi

    echo "$FICHIER_CSV"
}

# ── Nettoyer un fichier CSV ───────────────────────────────
function nettoyer_csv() {
    local TABLE="$1"
    local FICHIER_CSV="$2"

    if [ ! -f "$FICHIER_CSV" ]; then
        log_message "ERROR" "Fichier CSV introuvable : $FICHIER_CSV"
        return
    fi

    local AVANT=$(wc -l < "$FICHIER_CSV")
    local HEADER=$(head -1 "$FICHIER_CSV")
    local NB_COLS=$(echo "$HEADER" | awk -F',' '{print NF}')

    # Étape 1 — Supprimer les lignes vides
    grep -v '^$' "$FICHIER_CSV" > /tmp/step1_${TABLE}.csv

    # Étape 2 — Supprimer les doublons en gardant l'en-tête
    echo "$HEADER" > /tmp/step2_${TABLE}.csv
    tail -n +2 /tmp/step1_${TABLE}.csv | sort | uniq >> /tmp/step2_${TABLE}.csv

    # Étape 3 — Remplacer les valeurs manquantes
    cp /tmp/step2_${TABLE}.csv /tmp/step3_${TABLE}.csv

    for COL in $(seq 1 $NB_COLS); do

        # Extraire toutes les valeurs non nulles de cette colonne
        VALEURS=$(awk -F',' -v col=$COL 'NR>1 {
            gsub(/^[[:space:]]+|[[:space:]]+$/, "", $col)
            if($col != "" && $col != "null" && $col != "NULL" && $col != "\\N")
                print $col
        }' /tmp/step2_${TABLE}.csv)

        if [ -z "$VALEURS" ]; then
            continue
        fi

        # Détecter si la colonne est numérique ou string
        EST_NUMERIQUE=$(echo "$VALEURS" | head -5 | \
            awk '{if($0 ~ /^-?[0-9]+(\.[0-9]+)?$/) print "oui"; else print "non"}' | \
            sort | uniq -c | sort -rn | head -1 | awk '{print $2}')

        if [ "$EST_NUMERIQUE" = "oui" ]; then
            # Calculer la moyenne
            REMPLACEMENT=$(echo "$VALEURS" | \
                awk '{sum+=$1; count++} END {printf "%.2f", sum/count}')
            log_message "INFOS" "Colonne $COL ($TABLE) numérique → moyenne : $REMPLACEMENT"
        else
            # Calculer la valeur la plus fréquente
            REMPLACEMENT=$(echo "$VALEURS" | \
                sort | uniq -c | sort -rn | head -1 | awk '{print $2}')
            log_message "INFOS" "Colonne $COL ($TABLE) string → plus fréquente : $REMPLACEMENT"
        fi

        # Remplacer les valeurs manquantes dans cette colonne
        awk -F',' -v col=$COL -v val="$REMPLACEMENT" 'BEGIN{OFS=","} {
            if(NR > 1) {
                gsub(/^[[:space:]]+|[[:space:]]+$/, "", $col)
                if($col == "" || $col == "null" || $col == "NULL" || $col == "\\N")
                    $col = val
            }
            print
        }' /tmp/step3_${TABLE}.csv > /tmp/step3_tmp_${TABLE}.csv

        mv /tmp/step3_tmp_${TABLE}.csv /tmp/step3_${TABLE}.csv

    done

    local APRES=$(wc -l < /tmp/step3_${TABLE}.csv)
    local SUPPRIMES=$((AVANT - APRES))

    cp /tmp/step3_${TABLE}.csv "/tmp/mindctl_${TABLE}_clean.csv"

    echo "  ✅ $TABLE → $SUPPRIMES doublons supprimés — valeurs manquantes remplacées"
    log_message "INFOS" "Table $TABLE nettoyée avec succès"

    rm -f /tmp/step1_${TABLE}.csv \
          /tmp/step2_${TABLE}.csv \
          /tmp/step3_${TABLE}.csv \
          /tmp/step3_tmp_${TABLE}.csv
}

# ── Lancer le nettoyage ───────────────────────────────────
echo ""
echo "[cleaner] Récupération des tables..."
TABLES=$(get_tables)

if [ -z "$TABLES" ]; then
    log_message "ERROR" "Aucune table trouvée"
    erreur 107
fi

echo "[cleaner] Tables trouvées : $TABLES"
echo ""
echo "[cleaner] Nettoyage en cours..."
echo ""

TOTAL_TABLES=0

for TABLE in $TABLES; do
    FICHIER_CSV=$(extraire_table "$TABLE")
    nettoyer_csv "$TABLE" "$FICHIER_CSV"
    TOTAL_TABLES=$((TOTAL_TABLES + 1))
done

echo ""
echo "[cleaner] ── Résumé ──"
echo "[cleaner] $TOTAL_TABLES tables nettoyées ✅"
echo "[cleaner] Fichiers propres → /tmp/mindctl_*_clean.csv"
log_message "INFOS" "cleaner terminé — $TOTAL_TABLES tables traitées"
```

## 🔍 Explication ligne par ligne

---

### Imports et contexte

`source "$CONTEXT_FILE"` → charge les variables `DB_TYPE`, `DB_PORT`, `DB_USER`, `DB_PASS`, `DB_NAME` écrites par depguard. Sans ça cleaner ne sait pas à quelle base se connecter.

`[ ! -f "$CONTEXT_FILE" ]` → `!` = NOT, `-f` = fichier existe. Si le contexte n'existe pas → depguard n'a pas encore tourné → erreur 102

`[ -z "$DB_TYPE" ]` → `-z` = vide. Si aucune base détectée dans le contexte → erreur 107

---

### get_tables()

```
PGPASSWORD=$DB_PASS psql ... -c "SELECT tablename FROM pg_tables WHERE schemaname='public';"
```

`PGPASSWORD=$DB_PASS` → passer le mot de passe à psql sans qu'il le demande interactivement

`-h localhost` → adresse du serveur

`-p $DB_PORT` → port de connexion

`-U $DB_USER` → utilisateur

`-d $DB_NAME` → nom de la base

`-t` → tuple only → afficher seulement les données sans les en-têtes

`-c "..."` → exécuter cette requête SQL directement

`WHERE schemaname='public'` → prendre uniquement les tables publiques, pas les tables système

`2>/dev/null` → cacher les messages d'erreur

`grep -v '^$'` → supprimer les lignes vides du résultat

`xargs` → mettre tout sur une seule ligne séparée par des espaces

**Pour MySQL :**

`SHOW TABLES;` → commande SQL pour lister toutes les tables

`tail -n +2` → ignorer la première ligne qui est l'en-tête `Tables_in_nombase`

---

### extraire_table()

```
local TABLE="$1"
local FICHIER_CSV="/tmp/mindctl_${TABLE}.csv"
```

`local` → variable locale à la fonction

`/tmp/mindctl_${TABLE}.csv` → Ex : table `users` → fichier `/tmp/mindctl_users.csv`

**Pour PostgreSQL :**

```
\COPY $TABLE TO '$FICHIER_CSV' CSV HEADER
```

`\COPY` → commande psql pour exporter une table entière en CSV. `HEADER` = inclure les noms des colonnes sur la première ligne

**Pour MySQL :**

```
SELECT * FROM $TABLE INTO OUTFILE '$FICHIER_CSV'
FIELDS TERMINATED BY ','
ENCLOSED BY '"'
LINES TERMINATED BY '\n';
```

`INTO OUTFILE` → exporter directement dans un fichier

`FIELDS TERMINATED BY ','` → séparer les colonnes par des virgules

`ENCLOSED BY '"'` → entourer les valeurs de guillemets

`echo "$FICHIER_CSV"` → retourner le chemin du fichier pour que la fonction appelante puisse l'utiliser

---

### nettoyer_csv()

```
local AVANT=$(wc -l < "$FICHIER_CSV")
```

`wc -l` → compter le nombre de lignes. `<` = lire depuis le fichier sans afficher le nom

```
local HEADER=$(head -1 "$FICHIER_CSV")
local NB_COLS=$(echo "$HEADER" | awk -F',' '{print NF}')
```

`head -1` → lire uniquement la première ligne = noms des colonnes

`awk -F','` → définir `,` comme séparateur de colonnes

`{print NF}` → NF = Number of Fields = nombre de colonnes

---

**Étape 1 — Lignes vides :**

```
grep -v '^$' "$FICHIER_CSV" > /tmp/step1_${TABLE}.csv
```

`-v` → inverser → garder les lignes qui NE correspondent PAS au pattern

`'^$'` → ligne qui commence et finit immédiatement = ligne vide

---

**Étape 2 — Doublons :**

```
echo "$HEADER" > /tmp/step2_${TABLE}.csv
tail -n +2 /tmp/step1_${TABLE}.csv | sort | uniq >> /tmp/step2_${TABLE}.csv
```

`echo "$HEADER" >` → écrire l'en-tête en premier dans le nouveau fichier

`tail -n +2` → ignorer la ligne 1 (en-tête) et prendre à partir de la ligne 2

`sort` → trier les lignes — obligatoire car `uniq` ne fonctionne que sur des lignes consécutives identiques

`uniq` → supprimer les lignes consécutives identiques = les doublons

`>>` → ajouter à la fin du fichier sans écraser l'en-tête déjà écrit

---

**Étape 3 — Valeurs manquantes :**

```
for COL in $(seq 1 $NB_COLS); do
```

`seq 1 $NB_COLS` → générer la séquence 1 2 3 ... jusqu'au nombre de colonnes. Ex : 4 colonnes → `1 2 3 4`

```
VALEURS=$(awk -F',' -v col=$COL 'NR>1 {
    gsub(/^[[:space:]]+|[[:space:]]+$/, "", $col)
    if($col != "" && $col != "null" && $col != "NULL" && $col != "\\N")
        print $col
}' /tmp/step2_${TABLE}.csv)
```

`-v col=$COL` → passer la variable Bash `$COL` à awk

`NR>1` → ignorer la ligne 1 (en-tête), traiter à partir de la ligne 2

`gsub(/^[[:space:]]+|[[:space:]]+$/, "", $col)` → supprimer les espaces au début et à la fin de la valeur

`$col` → la valeur de la colonne numéro `col`

`\\N` → représentation MySQL des valeurs nulles

```
if [ -z "$VALEURS" ]; then
    continue
fi
```

Si toute la colonne est nulle → `continue` → passer à la colonne suivante sans rien faire

---

**Détecter si numérique ou string :**

```
EST_NUMERIQUE=$(echo "$VALEURS" | head -5 | \
    awk '{if($0 ~ /^-?[0-9]+(\.[0-9]+)?$/) print "oui"; else print "non"}' | \
    sort | uniq -c | sort -rn | head -1 | awk '{print $2}')
```

`head -5` → regarder seulement les 5 premières valeurs pour décider

`/^-?[0-9]+(\.[0-9]+)?$/` → expression régulière qui détecte un nombre :

```
^        → début de ligne
-?       → signe moins optionnel (nombres négatifs)
[0-9]+   → un ou plusieurs chiffres
(\.[0-9]+)? → partie décimale optionnelle
$        → fin de ligne
```

`sort | uniq -c | sort -rn` → compter les `oui` et `non`, trier du plus fréquent au moins fréquent

`head -1 | awk '{print $2}'` → prendre la première ligne = le type le plus fréquent

---

**Calculer la moyenne pour les numériques :**

```
REMPLACEMENT=$(echo "$VALEURS" | \
    awk '{sum+=$1; count++} END {printf "%.2f", sum/count}')
```

`sum+=$1` → additionner chaque valeur à la somme

`count++` → compter le nombre de valeurs

`END {...}` → exécuter après avoir lu toutes les lignes

`printf "%.2f"` → afficher avec 2 décimales. Ex : `42.67`

---

**Calculer la valeur la plus fréquente pour les strings :**

```
REMPLACEMENT=$(echo "$VALEURS" | \
    sort | uniq -c | sort -rn | head -1 | awk '{print $2}')
```

`sort` → trier pour regrouper les identiques

`uniq -c` → compter les occurrences de chaque valeur

`sort -rn` → trier par nombre décroissant (`-r` = reverse, `-n` = numérique)

`head -1` → prendre la première = la plus fréquente

`awk '{print $2}'` → la colonne 2 = la valeur (colonne 1 = le compteur)

---

**Remplacer les nulls dans le fichier :**

```
awk -F',' -v col=$COL -v val="$REMPLACEMENT" 'BEGIN{OFS=","} {
    if(NR > 1) {
        gsub(/^[[:space:]]+|[[:space:]]+$/, "", $col)
        if($col == "" || $col == "null" || $col == "NULL" || $col == "\\N")
            $col = val
    }
    print
}' /tmp/step3_${TABLE}.csv > /tmp/step3_tmp_${TABLE}.csv
```

`-v val="$REMPLACEMENT"` → passer la valeur de remplacement à awk

`BEGIN{OFS=","}` → définir `,` comme séparateur de sortie pour reconstruire le CSV

`$col = val` → remplacer la valeur nulle par la valeur calculée

`print` → réécrire la ligne avec la valeur remplacée

```
mv /tmp/step3_tmp_${TABLE}.csv /tmp/step3_${TABLE}.csv
```

`mv` → renommer le fichier temporaire. On ne peut pas écrire dans le fichier qu'on lit en même temps avec awk — on écrit dans un temporaire puis on le renomme

---

### Nettoyage final

```
rm -f /tmp/step1_${TABLE}.csv \
      /tmp/step2_${TABLE}.csv \
      /tmp/step3_${TABLE}.csv \
      /tmp/step3_tmp_${TABLE}.csv
```

`rm -f` → supprimer les fichiers temporaires. `-f` = force, ne pas afficher d'erreur si le fichier n'existe pas

---

### Boucle principale

```
for TABLE in $TABLES; do
    FICHIER_CSV=$(extraire_table "$TABLE")
    nettoyer_csv "$TABLE" "$FICHIER_CSV"
    TOTAL_TABLES=$((TOTAL_TABLES + 1))
done
```

`$(extraire_table "$TABLE")` → appeler la fonction et récupérer le chemin du CSV retourné par `echo`

`$((TOTAL_TABLES + 1))` → `$((...))` = calcul arithmétique en Bash
