## 🎯 Ce que fait ce fichier

Lit les fichiers CSV nettoyés par `cleaner.sh`, calcule des statistiques sur chaque colonne, détecte les anomalies, et envoie tout au LLM pour interprétation intelligente.

```
Pour chaque table nettoyée :
    Pour chaque colonne :
        Si numérique → min, max, moyenne, médiane
        Si texte     → valeur la plus/moins fréquente, uniques
    Détecter les lignes avec trop de nulls
    Envoyer les stats au LLM → recommandations
Sauvegarder pour insight.sh
```

> [!info] Lit les fichiers `/tmp/mindctl_*_clean.csv` produits par `cleaner.sh` Sauvegarde dans `/var/log/mindctl/reports/last_data.txt` pour `insight.sh`

## 💻 Le code complet

```bash
#!/bin/bash

source ~/mindctl/mindctl.conf
source ~/mindctl/core/logger.sh
source ~/mindctl/core/errors.sh
source ~/mindctl/core/llm.sh

CONTEXT_FILE="/tmp/mindctl_context.env"

# ── Vérifier le contexte depguard ────────────────────────
if [ ! -f "$CONTEXT_FILE" ]; then
    log_message "ERROR" "Contexte depguard manquant"
    erreur 102
fi

source "$CONTEXT_FILE"

if [ -z "$DB_TYPE" ]; then
    log_message "ERROR" "Aucune base de données détectée"
    erreur 107
fi

log_message "INFOS" "analyzer démarré — base : $DB_TYPE"
echo ""
echo "[analyzer] ── Analyse statistique des données ──"

# ── Récupérer les tables nettoyées ────────────────────────
function get_tables_propres() {
    ls /tmp/mindctl_*_clean.csv 2>/dev/null | \
        sed 's|/tmp/mindctl_||' | \
        sed 's|_clean.csv||'
}

# ── Analyser une table ────────────────────────────────────
function analyser_table() {
    local TABLE="$1"
    local FICHIER="/tmp/mindctl_${TABLE}_clean.csv"

    if [ ! -f "$FICHIER" ]; then
        log_message "ERROR" "Fichier propre introuvable : $TABLE"
        return
    fi

    local HEADER=$(head -1 "$FICHIER")
    local NB_COLS=$(echo "$HEADER" | awk -F',' '{print NF}')
    local NB_LIGNES=$(tail -n +2 "$FICHIER" | wc -l)

    echo ""
    echo "[analyzer] Table : $TABLE ($NB_LIGNES lignes, $NB_COLS colonnes)"
    echo "─────────────────────────────────"

    STATS=""

    # ── Analyser chaque colonne ───────────────────────────
    for COL in $(seq 1 $NB_COLS); do

        NOM_COL=$(echo "$HEADER" | awk -F',' -v c=$COL '{print $c}' | xargs)

        VALEURS=$(awk -F',' -v col=$COL 'NR>1 {print $col}' "$FICHIER" | \
                  grep -v '^$')

        NB_TOTAL=$(echo "$VALEURS" | wc -l)
        NB_UNIQUES=$(echo "$VALEURS" | sort | uniq | wc -l)
        NB_NULLS=$(awk -F',' -v col=$COL 'NR>1 {
            if($col=="" || $col=="null" || $col=="NULL")
                count++
        } END {print count+0}' "$FICHIER")

        # Détecter si numérique
        EST_NUM=$(echo "$VALEURS" | head -5 | \
            awk '{if($0~/^-?[0-9]+(\.[0-9]+)?$/) print "oui"; else print "non"}' | \
            sort | uniq -c | sort -rn | head -1 | awk '{print $2}')

        if [ "$EST_NUM" = "oui" ]; then
            STATS_NUM=$(echo "$VALEURS" | awk '
                BEGIN { min=999999999; max=-999999999 }
                {
                    sum += $1
                    count++
                    if($1 < min) min = $1
                    if($1 > max) max = $1
                    vals[count] = $1
                }
                END {
                    moyenne = sum/count
                    n = asort(vals)
                    if(n%2==0) mediane = (vals[n/2]+vals[n/2+1])/2
                    else mediane = vals[int(n/2)+1]
                    printf "min=%.2f max=%.2f moy=%.2f med=%.2f", min, max, moyenne, mediane
                }')

            echo "  📊 $NOM_COL (numérique) : $STATS_NUM | nulls=$NB_NULLS | uniques=$NB_UNIQUES"
            STATS="$STATS\nColonne $NOM_COL : $STATS_NUM | nulls=$NB_NULLS"

        else
            PLUS_FREQ=$(echo "$VALEURS" | sort | uniq -c | \
                        sort -rn | head -1 | awk '{print $2}')
            MOINS_FREQ=$(echo "$VALEURS" | sort | uniq -c | \
                         sort -n | head -1 | awk '{print $2}')

            echo "  📝 $NOM_COL (texte) : plus_fréquent=$PLUS_FREQ | moins_fréquent=$MOINS_FREQ | uniques=$NB_UNIQUES | nulls=$NB_NULLS"
            STATS="$STATS\nColonne $NOM_COL : plus_fréquent=$PLUS_FREQ | uniques=$NB_UNIQUES | nulls=$NB_NULLS"
        fi

    done

    # ── Détecter les anomalies ────────────────────────────
    echo ""
    echo "[analyzer] Détection des anomalies..."

    ANOMALIES=$(awk -F',' -v cols=$NB_COLS 'NR>1 {
        null_count=0
        for(i=1; i<=cols; i++) {
            if($i=="" || $i=="null" || $i=="NULL") null_count++
        }
        if(null_count > cols/2)
            print "Ligne " NR " : " null_count "/" cols " valeurs nulles"
    }' "$FICHIER" | head -10)

    if [ -n "$ANOMALIES" ]; then
        echo "$ANOMALIES"
        STATS="$STATS\nAnomalies détectées :\n$ANOMALIES"
    else
        echo "  ✅ Aucune anomalie critique détectée"
    fi

    # ── Sauvegarder les stats ─────────────────────────────
    echo -e "$STATS" >> /var/log/mindctl/reports/last_data.txt
    log_message "INFOS" "Statistiques $TABLE sauvegardées"

    # ── Envoyer au LLM pour interprétation ───────────────
    echo ""
    echo "[analyzer] Interprétation LLM..."

    PROMPT="Voici les statistiques de la table '$TABLE' en production :
$STATS

Analyse ces statistiques et réponds en français :
1. Y a-t-il des anomalies critiques ?
2. Quelle colonne pose le plus de problème ?
3. Quelles sont tes recommandations concrètes ?
Sois concis — maximum 5 lignes."

    INTERPRETATION=$(appeler_llm "$PROMPT")

    echo ""
    echo "[analyzer] Interprétation :"
    echo "─────────────────────────────────"
    echo "$INTERPRETATION"
    echo "─────────────────────────────────"

    log_message "INFOS" "Analyse table $TABLE terminée"
}

# ── Lancer l'analyse ──────────────────────────────────────
echo ""
echo "[analyzer] Récupération des fichiers nettoyés..."

TABLES=$(get_tables_propres)

if [ -z "$TABLES" ]; then
    log_message "ERROR" "Aucun fichier nettoyé — lancez cleaner d'abord"
    erreur 107
fi

echo "[analyzer] Tables à analyser : $TABLES"

> /var/log/mindctl/reports/last_data.txt

TOTAL=0
for TABLE in $TABLES; do
    analyser_table "$TABLE"
    TOTAL=$((TOTAL + 1))
done

echo ""
echo "[analyzer] ── Résumé ──"
echo "[analyzer] $TOTAL tables analysées ✅"
echo "[analyzer] Rapport → /var/log/mindctl/reports/last_data.txt"
log_message "INFOS" "analyzer terminé — $TOTAL tables analysées"
```

## 🔍 Explication ligne par ligne

---

### Imports et vérifications

`source ~/mindctl/core/llm.sh` → charge `appeler_llm()` car analyzer envoie les statistiques au LLM pour interprétation

`source "$CONTEXT_FILE"` → charge `DB_TYPE`, `DB_PORT`... écrits par depguard

`[ ! -f "$CONTEXT_FILE" ]` → si le contexte n'existe pas → depguard n'a pas tourné → erreur 102

`[ -z "$DB_TYPE" ]` → si aucune base détectée → erreur 107

---

### get_tables_propres()

```
ls /tmp/mindctl_*_clean.csv 2>/dev/null
```

`ls` → lister les fichiers. `*` = n'importe quoi. Liste tous les fichiers nettoyés par cleaner.

`2>/dev/null` → si aucun fichier trouvé → cacher le message d'erreur de ls

```
sed 's|/tmp/mindctl_||'
sed 's|_clean.csv||'
```

Deux `sed` pour extraire uniquement le nom de la table depuis le chemin complet :

```
/tmp/mindctl_users_clean.csv
→ après premier sed  : users_clean.csv
→ après deuxième sed : users
```

---

### analyser_table()

```
local HEADER=$(head -1 "$FICHIER")
local NB_COLS=$(echo "$HEADER" | awk -F',' '{print NF}')
local NB_LIGNES=$(tail -n +2 "$FICHIER" | wc -l)
```

`head -1` → lire la première ligne = noms des colonnes

`awk -F',' '{print NF}'` → NF = Number of Fields = compter les colonnes en se basant sur les virgules

`tail -n +2` → ignorer la première ligne (en-tête) et compter seulement les données

---

### Boucle par colonne

```
for COL in $(seq 1 $NB_COLS); do
```

`seq 1 $NB_COLS` → générer la séquence 1 2 3... Ex : 4 colonnes → `1 2 3 4`

```
NOM_COL=$(echo "$HEADER" | awk -F',' -v c=$COL '{print $c}' | xargs)
```

`-v c=$COL` → passer le numéro de colonne à awk

`{print $c}` → afficher la colonne numéro `c` de l'en-tête = nom de la colonne

`xargs` → supprimer les espaces

```
VALEURS=$(awk -F',' -v col=$COL 'NR>1 {print $col}' "$FICHIER" | grep -v '^$')
```

Extraire toutes les valeurs de la colonne numéro `col` en ignorant l'en-tête (`NR>1`)

`grep -v '^$'` → supprimer les lignes vides du résultat

```
NB_UNIQUES=$(echo "$VALEURS" | sort | uniq | wc -l)
```

`sort | uniq` → trier et dédupliquer → `wc -l` compter = nombre de valeurs uniques

```
NB_NULLS=$(awk -F',' -v col=$COL 'NR>1 {
    if($col=="" || $col=="null" || $col=="NULL")
        count++
} END {print count+0}' "$FICHIER")
```

Compter les valeurs nulles dans cette colonne. `count+0` → si count n'a jamais été incrémenté → afficher `0` au lieu de rien

---

### Statistiques numériques

```
STATS_NUM=$(echo "$VALEURS" | awk '
    BEGIN { min=999999999; max=-999999999 }
    {
        sum += $1
        count++
        if($1 < min) min = $1
        if($1 > max) max = $1
        vals[count] = $1
    }
    END {
        moyenne = sum/count
        n = asort(vals)
        if(n%2==0) mediane = (vals[n/2]+vals[n/2+1])/2
        else mediane = vals[int(n/2)+1]
        printf "min=%.2f max=%.2f moy=%.2f med=%.2f", min, max, moyenne, mediane
    }')
```

`BEGIN { min=999999999; max=-999999999 }` → initialiser min très grand et max très petit pour que les vraies valeurs les remplacent dès le début

`sum += $1` → additionner chaque valeur

`vals[count] = $1` → stocker chaque valeur dans un tableau pour calculer la médiane

`asort(vals)` → trier le tableau et retourner le nombre d'éléments

**Calcul de la médiane :**

```
Si nombre pair   → médiane = moyenne des deux valeurs du milieu
Si nombre impair → médiane = valeur du milieu
```

`printf "min=%.2f ..."` → afficher avec 2 décimales

---

### Statistiques texte

```
PLUS_FREQ=$(echo "$VALEURS" | sort | uniq -c | sort -rn | head -1 | awk '{print $2}')
MOINS_FREQ=$(echo "$VALEURS" | sort | uniq -c | sort -n | head -1 | awk '{print $2}')
```

`uniq -c` → compter les occurrences de chaque valeur unique

`sort -rn` → trier par nombre décroissant (`-r` = reverse, `-n` = numérique) → plus fréquent en premier

`sort -n` → trier par nombre croissant → moins fréquent en premier

`awk '{print $2}'` → colonne 1 = compteur, colonne 2 = la valeur

---

### Détection des anomalies

```
ANOMALIES=$(awk -F',' -v cols=$NB_COLS 'NR>1 {
    null_count=0
    for(i=1; i<=cols; i++) {
        if($i=="" || $i=="null" || $i=="NULL") null_count++
    }
    if(null_count > cols/2)
        print "Ligne " NR " : " null_count "/" cols " valeurs nulles"
}' "$FICHIER" | head -10)
```

`for(i=1; i<=cols; i++)` → boucle awk sur toutes les colonnes d'une ligne

`null_count > cols/2` → si plus de la moitié des colonnes sont nulles → anomalie critique

`print "Ligne " NR " : ..."` → NR = numéro de ligne actuel dans awk

`head -10` → afficher seulement les 10 premières anomalies pour ne pas noyer le terminal

---

### Sauvegarde pour insight

```
echo -e "$STATS" >> /var/log/mindctl/reports/last_data.txt
```

`echo -e` → interpréter les `\n` comme de vrais sauts de ligne

`>>` → ajouter à la fin du fichier sans écraser — accumule les stats de toutes les tables

---

### Appel LLM

```
PROMPT="Voici les statistiques de la table '$TABLE' en production :
$STATS
...
Sois concis — maximum 5 lignes."
```

On demande au LLM de rester concis — 5 lignes maximum — pour ne pas noyer l'utilisateur dans du texte.

```
INTERPRETATION=$(appeler_llm "$PROMPT")
```

Appelle `appeler_llm()` de `core/llm.sh` et stocke la réponse dans `INTERPRETATION`

---

### Boucle principale

```
> /var/log/mindctl/reports/last_data.txt
```

`>` seul sans texte → vider le fichier avant de commencer. Comme ça les stats de la session précédente ne se mélangent pas avec les nouvelles.

```
TOTAL=$((TOTAL + 1))
```

`$((...))` → calcul arithmétique en Bash
