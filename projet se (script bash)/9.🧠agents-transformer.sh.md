```bash
#!/bin/bash

source ~/mindctl/mindctl.conf
source ~/mindctl/core/logger.sh
source ~/mindctl/core/errors.sh

FICHIER="$1"

# ── Vérifier le fichier ───────────────────────────────────
if [ -z "$FICHIER" ]; then
    log_message "ERROR" "Aucun fichier fourni"
    erreur 101
fi

if [ ! -f "$FICHIER" ]; then
    log_message "ERROR" "Fichier introuvable : $FICHIER"
    erreur 102
fi

# ── Détecter le format du fichier ────────────────────────
EXTENSION="${FICHIER##*.}"

log_message "INFOS" "transformer démarré — fichier : $FICHIER"
echo ""
echo "[transformer] ── Conversion de format ──"
echo "[transformer] Fichier : $FICHIER"
echo "[transformer] Format détecté : $EXTENSION"

# ── CSV vers JSON ─────────────────────────────────────────
function csv_vers_json() {
    local FICHIER_CSV="$1"
    local FICHIER_JSON="${FICHIER_CSV%.csv}.json"

    local HEADER=$(head -1 "$FICHIER_CSV")
    local NB_COLS=$(echo "$HEADER" | awk -F',' '{print NF}')

    # Extraire les noms des colonnes
    IFS=',' read -ra COLONNES <<< "$HEADER"

    echo "[" > "$FICHIER_JSON"

    local NB_LIGNES=$(tail -n +2 "$FICHIER_CSV" | wc -l)
    local COMPTEUR=0

    tail -n +2 "$FICHIER_CSV" | while IFS=',' read -ra VALEURS; do
        COMPTEUR=$((COMPTEUR + 1))
        echo "  {" >> "$FICHIER_JSON"

        for i in $(seq 0 $((NB_COLS - 1))); do
            local CLE="${COLONNES[$i]}"
            local VALEUR="${VALEURS[$i]}"

            # Supprimer les guillemets existants
            CLE=$(echo "$CLE" | tr -d '"' | xargs)
            VALEUR=$(echo "$VALEUR" | tr -d '"' | xargs)

            # Détecter si la valeur est numérique
            if echo "$VALEUR" | grep -qP '^-?[0-9]+(\.[0-9]+)?$'; then
                LIGNE="    \"$CLE\": $VALEUR"
            else
                LIGNE="    \"$CLE\": \"$VALEUR\""
            fi

            # Ajouter virgule si pas la dernière clé
            if [ $i -lt $((NB_COLS - 1)) ]; then
                echo "$LIGNE," >> "$FICHIER_JSON"
            else
                echo "$LIGNE" >> "$FICHIER_JSON"
            fi
        done

        # Ajouter virgule si pas le dernier objet
        if [ $COMPTEUR -lt $NB_LIGNES ]; then
            echo "  }," >> "$FICHIER_JSON"
        else
            echo "  }" >> "$FICHIER_JSON"
        fi
    done

    echo "]" >> "$FICHIER_JSON"

    echo "[transformer] ✅ CSV converti en JSON → $FICHIER_JSON"
    log_message "INFOS" "CSV vers JSON : $FICHIER_JSON"
}

# ── JSON vers CSV ─────────────────────────────────────────
function json_vers_csv() {
    local FICHIER_JSON="$1"
    local FICHIER_CSV="${FICHIER_JSON%.json}.csv"

    # Vérifier que jq est disponible
    if ! command -v jq &>/dev/null; then
        log_message "ERROR" "jq non installé"
        erreur 107
    fi

    # Extraire les clés du premier objet = noms des colonnes
    HEADER=$(jq -r '.[0] | keys | join(",")' "$FICHIER_JSON")
    echo "$HEADER" > "$FICHIER_CSV"

    # Extraire les valeurs de chaque objet
    jq -r '.[] | [.[]] | @csv' "$FICHIER_JSON" >> "$FICHIER_CSV"

    echo "[transformer] ✅ JSON converti en CSV → $FICHIER_CSV"
    log_message "INFOS" "JSON vers CSV : $FICHIER_CSV"
}

# ── Lancer la conversion ──────────────────────────────────
case "$EXTENSION" in
    csv)
        echo "[transformer] Conversion CSV → JSON..."
        csv_vers_json "$FICHIER"
        ;;
    json)
        echo "[transformer] Conversion JSON → CSV..."
        json_vers_csv "$FICHIER"
        ;;
    *)
        log_message "ERROR" "Format non supporté : $EXTENSION"
        erreur 104
        ;;
esac

log_message "INFOS" "transformer terminé"
```
